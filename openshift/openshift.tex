\newcommand{\layout}[3] {% {Id}{Size}{Location}
  \pgfmathsetmacro\layoutScale{#2}
  \pgfmathsetmacro\layoutSizeInner{#2}
  \pgfmathsetmacro\layoutSizeOuter{#2 + \layoutScale}
  \pgfmathsetmacro\layoutSizeFrame{#2 + \layoutScale*2}
  
  \def\layerList{layoutInner/\layoutSizeInner,
                 layoutOuter/\layoutSizeOuter,
                 layoutFrame/\layoutSizeFrame
                 }
                 
  \foreach \n/\s in \layerList {
    \node[circle, minimum size=\s cm, draw, dotted, opacity=0.5] (\n) {};
  }
}

\newcommand{\outline}[4] {% {Id}{Size}{Location}{outline}{draw}
  % Takes a list of outline shapes and sizes and draws them
  \foreach \shape/\height/\width/\id in \outlineList {
    \node[ \shape, 
           minimum size=\height cm,
           minimum height=\height cm, 
           minimum width=\width cm, %draw, 
           dotted
         ] (\id) {};
